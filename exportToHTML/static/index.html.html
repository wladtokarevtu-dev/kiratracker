<script>
    function sendWalk(person) {
        fetch('/walk?person=' + encodeURIComponent(person), {method: 'POST'})
            .then(() => loadStatus())
            .catch(() => {
                document.getElementById('info').innerText = 'Fehler beim Speichern.';
            });
    }

    function loadStatus() {
        fetch('/status')
            .then(r => r.json())
            .then(data => {
                const info = document.getElementById('info');

                if (!data.morning && !data.evening) {
                    info.innerText = 'Kira war heute noch nicht draußen.';
                } else if (data.morning && !data.evening) {
                    info.innerText = 'Kira war heute morgens draußen.';
                } else if (!data.morning && data.evening) {
                    info.innerText = 'Kira war heute abends draußen.';
                } else {
                    info.innerText = 'Kira war heute morgens und abends draußen.';
                }

                const last = data.entries.slice(-5).reverse();
                document.getElementById('lastWalks').innerText =
                    last.map(e => e.person + ' – ' + e.time).join('\n');

                const lb = document.getElementById('leaderboard');
                lb.innerHTML = '';
                Object.entries(data.leaderboard)
                    .sort((a, b) => b[1] - a[1])
                    .forEach(([name, count]) => {
                        const li = document.createElement('li');
                        li.textContent = name + ': ' + count;
                        lb.appendChild(li);
                    });
            })
            .catch(() => {
                document.getElementById('info').innerText = 'Fehler beim Laden des Status.';
            });
    }

    window.onload = loadStatus;
</script>
